const names = [
  'Atul',
  'Ed',
  'Fanni',
  'Gunjan',
  'Kava',
  'Pawel',
  'Ray',
  'Vinny',
  'Prudence'
];

const shuffle = (names) =>
  names
    .map((name) => ({ name, order: Math.random() }))
    .sort((a, b) => a.order - b.order)
    .map((person) => person.name)
    .join(' -> ');

shuffle(names);
